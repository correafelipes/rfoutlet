        const API_BASE = window.location.hostname === 'localhost' ? 'http://localhost:3001' : (window.location.hostname.includes('vercel.app') ? 'https://rfoutlet.onrender.com' : '');

        function toggleMobileMenu() {
            const sidebar = document.getElementById('sidebar');
            const overlay = document.getElementById('mobileOverlay');
            if (sidebar.classList.contains('hidden') || sidebar.classList.contains('-translate-x-full')) {
                sidebar.classList.remove('hidden', '-translate-x-full');
                sidebar.classList.add('flex', 'translate-x-0');
                overlay.classList.remove('hidden');
            } else {
                sidebar.classList.remove('translate-x-0');
                sidebar.classList.add('-translate-x-full');
                overlay.classList.add('hidden');
                setTimeout(() => {
                    if (window.innerWidth < 1024) sidebar.classList.add('hidden');
                }, 300);
            }
        }

        function showToast(title, detail, severity = 'success') {

            const toast = document.createElement('div');
            toast.className = `p-toast-message ${severity}`;
            toast.innerHTML = `
                <div>
                    <span class="block font-bold mb-1">${title}</span>
                    <span class="block text-sm">${detail}</span>
                </div>
            `;
            container.appendChild(toast);
            setTimeout(() => toast.remove(), 4000);
        }

            document.getElementById('userName').innerText = user.nome;
            document.getElementById('userRole').innerText = user.funcao;

        function logout() {
            localStorage.removeItem('rfUser');
            window.location.href = 'login.html';

        function showSection(name) {
            document.querySelectorAll('.content-section').forEach(s => s.classList.add('hidden'));
            document.getElementById('section-' + name).classList.remove('hidden');
            
            const titles = { 'dashboard': 'Dashboard (Vendas)', 'estoque': 'Gestão de Produtos', 'caixa': 'Financeiro', 'escala': 'Escala Mensal', 'redes': 'Redes Sociais', 'equipe': 'Gestão de Vendedores', 'lojas': 'Lojas e Franquias', 'aprovacoes': 'Aprovações', 'precificacao': 'Precificação' };
            document.getElementById('pageTitle').innerText = titles[name];
            
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('nav-btn-active'));
            event.currentTarget.classList.add('nav-btn-active');
            
            if (window.innerWidth < 1024) toggleMobileMenu();
        }

        function limparFiltros() {
            document.getElementById('filtroInicio').value = '';
            document.getElementById('filtroFim').value = '';
            loadRelatorios();
        }

        const formatMoney = (val) => "R$ " + (val || 0).toLocaleString('pt-BR', {minimumFractionDigits: 2});

        async function loadRelatorios() {
            try {
                const dataInicio = document.getElementById('filtroInicio').value;
                const dataFim = document.getElementById('filtroFim').value;
                let url = `${API_BASE}/api/relatorios/vendas`;
                if(dataInicio && dataFim) url += `?data_inicio=${dataInicio}&data_fim=${dataFim}`;

                const res = await fetch(url);
                const relatorio = await res.json();

                const tbodyMarcas = document.getElementById('rankingMarcas');
                tbodyMarcas.innerHTML = '';
                if(relatorio.por_marca.length === 0) tbodyMarcas.innerHTML = '<tr><td colspan="3" class="text-center text-slate-400">Nenhuma venda no período.</td></tr>';
                relatorio.por_marca.forEach(m => {
                    tbodyMarcas.innerHTML += `<tr>
                        <td class="font-bold text-slate-700">${m.marca}</td>
                        <td class="text-center">${m.qtd_total} un</td>
                        <td class="text-right text-indigo-600 font-bold">${formatMoney(m.valor_total)}</td>
                    </tr>`;
                });

                const tbodyVends = document.getElementById('rankingVendedores');
                tbodyVends.innerHTML = '';
                if(relatorio.por_vendedor.length === 0) tbodyVends.innerHTML = '<tr><td colspan="3" class="text-center text-slate-400">Nenhuma venda no período.</td></tr>';
                relatorio.por_vendedor.forEach(v => {
                    tbodyVends.innerHTML += `<tr>
                        <td class="font-bold text-slate-700">${v.vendedor_nome}</td>
                        <td class="text-center">${v.qtd_total} un</td>
                        <td class="text-right text-emerald-600 font-bold">${formatMoney(v.valor_total)}</td>
                    </tr>`;
                });

                if(relatorio.por_mes && relatorio.por_mes.length > 0) {
                    const mesAtual = relatorio.por_mes[0];
                    document.getElementById('mesAtualValor').innerText = formatMoney(mesAtual.total_mes);
                    if(mesAtual.comparativo) {
                        document.getElementById('mesAnteriorValor').innerText = formatMoney(mesAtual.comparativo.valor_anterior) + ` (${mesAtual.comparativo.mes_anterior})`;
                        let badgeColor = parseFloat(mesAtual.comparativo.porcentagem) >= 0 ? 'text-green-600 bg-green-100' : 'text-red-600 bg-red-100';
                        let sinal = parseFloat(mesAtual.comparativo.porcentagem) >= 0 ? '+' : '';
                        document.getElementById('mesAtualBadge').innerHTML = `<span class="${badgeColor} px-2 py-1 font-bold rounded">${sinal}${mesAtual.comparativo.porcentagem} em relação ao mês passado</span>`;
                    } else {
                        document.getElementById('mesAnteriorValor').innerText = 'R$ 0,00';
                        document.getElementById('mesAtualBadge').innerHTML = `<span class="text-slate-500">Sem dados para comparar.</span>`;
                    }
                }
            } catch (e) { console.error("Erro Relatorios", e); }
        }

        async function loadVendedores() {
            try {
                const res = await fetch(`${API_BASE}/api/vendedores`);
                const equipe = await res.json();
                const tbody = document.getElementById('equipeTable');
                tbody.innerHTML = '';
                equipe.forEach(v => {
                    tbody.innerHTML += `<tr>
                        <td>#${v.id}</td>
                        <td class="font-bold text-slate-800">${v.nome}</td>
                        <td>${v.login}</td>
                        <td><span class="bg-blue-100 text-blue-800 px-2 py-1 rounded text-xs font-bold">${v.funcao}</span></td>
                    </tr>`;
                });
            } catch (e) {}
        }

        async function cadastrarVendedor() {
            const nome = document.getElementById('novoNome').value;
            const login = document.getElementById('novoLogin').value;
            const senha = document.getElementById('novaSenha').value;
            if(!nome || !login || !senha) return showToast('Erro', 'Preencha todos os campos.', 'error');
            try {
                const res = await fetch(`${API_BASE}/api/usuarios`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ nome, login, senha, funcao: 'Vendedor' })
                });
                const data = await res.json();
                if(data.success) {
                    showToast('Sucesso', 'Vendedor cadastrado!');
                    document.getElementById('novoNome').value = '';
                    document.getElementById('novoLogin').value = '';
                    loadVendedores();
                } else showToast('Erro', data.error, 'error');
            } catch (e) {}
        }

        async function loadOutros() {
            try {
                const resEst = await fetch(`${API_BASE}/api/estoque`);
                const estoque = await resEst.json();
                const tableBody = document.getElementById('estoqueTable');
                tableBody.innerHTML = '';
                estoque.forEach(item => {
                    tableBody.innerHTML += `<tr>
                        <td class="font-semibold">${item.codigo}</td>
                        <td><div class="font-bold">${item.marca}</div><div class="text-xs text-slate-500">${item.categoria}</div></td>
                        <td>${item.tamanho}</td>
                        <td>${item.quantidade}</td>
                        <td class="font-bold">R$ ${item.preco.toFixed(2)}</td>
                    </tr>`;
                });

                const resCaixa = await fetch(`${API_BASE}/api/caixa`);
                const caixa = await resCaixa.json();
                const caixaBody = document.getElementById('caixaTable');
                caixaBody.innerHTML = '';
                caixa.forEach(mov => {
                    let badgeClass = mov.tipo === 'ENTRADA' ? 'text-green-700 bg-green-100' : 'text-red-700 bg-red-100';
                    caixaBody.innerHTML += `<tr>
                        <td><span class="${badgeClass} px-2 py-1 rounded text-xs font-bold">${mov.tipo}</span></td>
                        <td class="font-bold">R$ ${mov.valor.toFixed(2)}</td>
                        <td>${mov.descricao}</td>
                        <td class="text-sm text-slate-500">${new Date(mov.data).toLocaleString()}</td>
                    </tr>`;
                });
            } catch (err) {}
        }

        window.onload = () => {
            loadRelatorios();
            loadVendedores();
            loadOutros();
        };
    </script>
