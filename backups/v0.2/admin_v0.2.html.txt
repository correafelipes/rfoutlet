    <script>
        const API_BASE = window.location.hostname === 'localhost' ? 'http://localhost:3001' : (window.location.hostname.includes('vercel.app') ? 'https://rfoutlet.onrender.com' : '');
        let currentUser = null;

        function showToast(title, detail, severity = 'success') {
            const container = document.getElementById('p-toast-container');
            const toast = document.createElement('div');
            toast.className = `p-toast-message ${severity}`;
            toast.innerHTML = `<div><span class="block font-bold mb-1">${title}</span><span class="block text-sm">${detail}</span></div>`;
            container.appendChild(toast);
            setTimeout(() => toast.remove(), 4000);
        }

        const userStr = localStorage.getItem('rfUser');
        if (!userStr) window.location.href = 'login.html';
        else {
            currentUser = JSON.parse(userStr);
            document.getElementById('userName').innerText = currentUser.nome;
            document.getElementById('userRole').innerText = currentUser.funcao;
        }

        function logout() {
            localStorage.removeItem('rfUser');
            window.location.href = 'login.html';
        }

        function showSection(name) {
            document.querySelectorAll('.content-section').forEach(s => s.classList.add('hidden'));
            document.getElementById('section-' + name).classList.remove('hidden');
            
            const titles = { 
                'dashboard': 'Córtex (Painel Global)', 
                'estoque': 'Estoque Matriz (Loja)', 
                'caixa': 'Financeiro', 
                'equipe': 'Gestão de Vendedores',
                'representantes': 'Maleta do Representante',
                'loja_pdv': 'PDV & RH'
            };
            document.getElementById('pageTitle').innerText = titles[name];
            
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('nav-btn-active'));
            event.currentTarget.classList.add('nav-btn-active');
        }

        function limparFiltros() {
            document.getElementById('filtroInicio').value = '';
            document.getElementById('filtroFim').value = '';
            loadRelatorios();
        }

        const formatMoney = (val) => "R$ " + (val || 0).toLocaleString('pt-BR', {minimumFractionDigits: 2});

        async function loadRelatorios() {
            try {
                const dataInicio = document.getElementById('filtroInicio').value;
                const dataFim = document.getElementById('filtroFim').value;
                let url = `${API_BASE}/api/relatorios/vendas`;
                if(dataInicio && dataFim) url += `?data_inicio=${dataInicio}&data_fim=${dataFim}`;

                const res = await fetch(url);
                const relatorio = await res.json();

                // Pipeline
                document.getElementById('previsaoPipeline').innerText = formatMoney(relatorio.previsao_pipeline);

                const tbodyMarcas = document.getElementById('rankingMarcas');
                tbodyMarcas.innerHTML = '';
                relatorio.por_marca.forEach(m => {
                    tbodyMarcas.innerHTML += `<tr>
                        <td class="font-bold text-slate-700">${m.marca}</td>
                        <td class="text-center">${m.qtd_total} un</td>
                        <td class="text-right text-indigo-600 font-bold">${formatMoney(m.valor_total)}</td>
                    </tr>`;
                });

                const tbodyVends = document.getElementById('rankingVendedores');
                tbodyVends.innerHTML = '';
                relatorio.por_vendedor.forEach(v => {
                    tbodyVends.innerHTML += `<tr>
                        <td class="font-bold text-slate-700">${v.vendedor_nome}</td>
                        <td class="text-center">${v.qtd_total} un</td>
                        <td class="text-right text-emerald-600 font-bold">${formatMoney(v.valor_total)}</td>
                    </tr>`;
                });

                if(relatorio.por_mes && relatorio.por_mes.length > 0) {
                    const mesAtual = relatorio.por_mes[0];
                    document.getElementById('mesAtualValor').innerText = formatMoney(mesAtual.total_mes);
                    if(mesAtual.comparativo) {
                        document.getElementById('mesAnteriorValor').innerText = formatMoney(mesAtual.comparativo.valor_anterior);
                        let badgeColor = parseFloat(mesAtual.comparativo.porcentagem) >= 0 ? 'text-green-600 bg-green-100' : 'text-red-600 bg-red-100';
                        document.getElementById('mesAtualBadge').innerHTML = `<span class="${badgeColor} px-2 py-1 font-bold rounded">${mesAtual.comparativo.porcentagem} vs. mês anterior</span>`;
                    }
                }
            } catch (e) {}
        }

        async function loadVendedores() {
            try {
                const res = await fetch(`${API_BASE}/api/vendedores`);
                const equipe = await res.json();
                
                const tbody = document.getElementById('equipeTable');
                const repSelect = document.getElementById('repSelect');
                
                tbody.innerHTML = '';
                repSelect.innerHTML = '<option value="">Selecione...</option>';

                equipe.forEach(v => {
                    tbody.innerHTML += `<tr>
                        <td>#${v.id}</td>
                        <td class="font-bold">${v.nome}</td>
                        <td>${v.login}</td>
                        <td><span class="bg-blue-100 text-blue-800 px-2 py-1 rounded text-xs font-bold">${v.funcao}</span></td>
                    </tr>`;

                    if(v.funcao.includes('Representante')) {
                        repSelect.innerHTML += `<option value="${v.id}">${v.nome}</option>`;
                    }
                });
            } catch (e) {}
        }

        async function cadastrarVendedor() {
            const nome = document.getElementById('novoNome').value;
            const login = document.getElementById('novoLogin').value;
            const senha = document.getElementById('novaSenha').value;
            const funcao = document.getElementById('novoFuncao').value;

            if(!nome || !login || !senha) return showToast('Erro', 'Preencha todos os campos.', 'error');
            try {
                const res = await fetch(`${API_BASE}/api/usuarios`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ nome, login, senha, funcao })
                });
                const data = await res.json();
                if(data.success) {
                    showToast('Sucesso', 'Vendedor/Representante cadastrado!');
                    loadVendedores();
                } else showToast('Erro', data.error, 'error');
            } catch (e) {}
        }

        // PONTO RH
        async function registrarPonto(tipo) {
            try {
                const res = await fetch(`${API_BASE}/api/ponto`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ usuario_id: currentUser.id, usuario_nome: currentUser.nome, tipo })
                });
                const data = await res.json();
                if(data.success) showToast('Ponto Registrado', `Seu ponto de ${tipo} foi gravado.`, 'success');
            } catch (e) { showToast('Erro', 'Falha ao bater o ponto.', 'error'); }
        }

        // TRANSFERENCIA DE MALETA
        async function transferirParaRepresentante() {
            const representante_id = document.getElementById('repSelect').value;
            const codigo = document.getElementById('repProdCodigo').value;
            const quantidade_transferir = document.getElementById('repProdQtd').value;

            if(!representante_id || !codigo || !quantidade_transferir) return showToast('Erro', 'Preencha todos os campos da maleta', 'error');

            try {
                const res = await fetch(`${API_BASE}/api/estoque/transferir`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ representante_id, codigo, quantidade_transferir })
                });
                const data = await res.json();
                if(data.success) {
                    showToast('Transferência Concluída', data.message, 'success');
                    loadOutros(); // recarrega o estoque principal
                } else showToast('Erro', data.error, 'error');
            } catch (e) {}
        }

        async function loadEstoqueRep() {
            const representante_id = document.getElementById('repSelect').value;
            if(!representante_id) return showToast('Aviso', 'Selecione um Representante primeiro.', 'error');

            try {
                const res = await fetch(`${API_BASE}/api/estoque?local=REP_${representante_id}`);
                const estoque = await res.json();
                const tbody = document.getElementById('estoqueRepTable');
                tbody.innerHTML = '';
                
                if(estoque.length === 0) return tbody.innerHTML = '<tr><td colspan="5" class="text-center p-4">A maleta deste representante está vazia.</td></tr>';

                estoque.forEach(item => {
                    tbody.innerHTML += `<tr>
                        <td class="font-bold">${item.codigo}</td>
                        <td>${item.marca}</td>
                        <td>${item.tamanho}</td>
                        <td class="text-orange-600 font-bold">${item.quantidade} un</td>
                        <td>R$ ${item.preco.toFixed(2)}</td>
                    </tr>`;
                });
            } catch (e) {}
        }

        async function loadOutros() {
            try {
                // Estoque só da loja
                const resEst = await fetch(`${API_BASE}/api/estoque?local=LOJA`);
                const estoque = await resEst.json();
                const tableBody = document.getElementById('estoqueTable');
                tableBody.innerHTML = '';
                estoque.forEach(item => {
                    tableBody.innerHTML += `<tr>
                        <td class="font-semibold">${item.codigo}</td>
                        <td><div class="font-bold">${item.marca}</div><div class="text-xs text-slate-500">${item.categoria}</div></td>
                        <td>${item.tamanho}</td>
                        <td class="font-bold">${item.quantidade}</td>
                        <td>R$ ${item.preco.toFixed(2)}</td>
                    </tr>`;
                });

                const resCaixa = await fetch(`${API_BASE}/api/caixa`);
                const caixa = await resCaixa.json();
                const caixaBody = document.getElementById('caixaTable');
                caixaBody.innerHTML = '';
                caixa.forEach(mov => {
                    let badgeClass = mov.tipo === 'ENTRADA' ? 'text-green-700 bg-green-100' : 'text-red-700 bg-red-100';
                    caixaBody.innerHTML += `<tr>
                        <td><span class="${badgeClass} px-2 py-1 rounded text-xs font-bold">${mov.tipo}</span></td>
                        <td class="font-bold">R$ ${mov.valor.toFixed(2)}</td>
                        <td>${mov.descricao}</td>
                        <td class="text-sm text-slate-500">${new Date(mov.data).toLocaleString()}</td>
                    </tr>`;
                });
            } catch (err) {}
        }

        window.onload = () => {

            loadVendedores();
            loadOutros();
